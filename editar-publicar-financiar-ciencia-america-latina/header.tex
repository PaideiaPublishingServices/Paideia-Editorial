% Comando para DOI del capítulo (forzando espaciado pequeño)
\newcommand{\chapterdoi}[1]{%
  \vspace*{-0.5em}  % Usar \vspace* para forzar
  \begin{center}
    \textcolor{chaptercolor}{\rule{0.3\textwidth}{0.5pt}}\\[-0.2em]  % Espaciado negativo
    \small\textbf{\color{chaptercolor}DOI:} \textrm{#1}\\[-0.2em]    % Espaciado negativo
    \textcolor{chaptercolor}{\rule{0.3\textwidth}{0.5pt}}
  \end{center}
  \vspace*{-0.3em}  % Espaciado negativo y forzado
}

% Comando personalizado para autores (forzando espaciado pequeño)
\newcommand{\chapterauthors}[1]{%
  \vspace*{-0.5em}  % Reducir espacio antes
  \begin{center}
    \Large\textit{\color{chaptercolor}#1}
  \end{center}
  \vspace*{-0.5em}  % Reducir espacio después
}

% Comando para afiliaciones (forzando espaciado pequeño)
\newcommand{\chapteraffiliations}[1]{%
  \vspace*{-0.3em}  % Reducir espacio antes
  \begin{center}
    \small\color{gray}#1
  \end{center}
  \vspace*{-0.2em}  % Espaciado muy pequeño
  \noindent\textcolor{chaptercolor}{\rule{\textwidth}{0.8pt}}
  \vspace*{0.2em}   % Espaciado mínimo después de línea
  \newpage
}
